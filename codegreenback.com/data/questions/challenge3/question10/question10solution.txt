def isPossible(ar, n) :
	sorted(ar)
	no=len(ar) 
	for i in  range(no) :
		ele=n-ar[i]
		if triplet(ar ,i+1 ,ele) :
			return "Yes"		
	return "No"
		
def triplet(ar , i , x) :
	j=len(ar)-1
	while (i<j) :
		if (x<ar[i]+ar[j]) :
			j-=1
		elif (x>ar[i]+ar[j]) :
			i+=1
		else :
			return True
	return False


if __name__=="__main__" :
	no=int(input())
	ar=[int(s) for s in input().split() ]
	n=int(input())
	print(isPossible(ar ,n))
